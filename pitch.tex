% Slide 1: Motivation and Problem Statement
\begin{frame}{Motivation and Problem Statement}
\textbf{Challenge:} Cryptocurrency markets exhibit high volatility and uncertainty
\vspace{0.5cm}

\textbf{Limitations of Traditional Approaches:}
\begin{itemize}
    \item Point estimates fail to capture uncertainty
    \item Frequentist confidence intervals lack intuitive interpretation
    \item No principled way to incorporate prior knowledge
\end{itemize}
\vspace{0.5cm}

\textbf{Our Solution:} Bayesian inference with Rejection Sampling
\begin{itemize}
    \item Complete probability distributions for parameters
    \item Credible intervals with direct probabilistic interpretation
    \item Posterior predictive distributions for forecasting
\end{itemize}
\end{frame}

% Slide 2: Methodology - Bayesian Model
\begin{frame}{Bayesian Model Specification}
\textbf{Likelihood:} Log returns follow a Normal distribution
\begin{equation*}
r_t = \log\left(\frac{P_t}{P_{t-1}}\right) \sim \mathcal{N}(\mu, \sigma^2)
\end{equation*}

\textbf{Prior Distributions:} Non-informative conjugate priors
\begin{align*}
\mu &\sim \mathcal{N}(\mu_0 = 0, \sigma_0^2 = 1000) \\
\sigma^2 &\sim \text{Inv-Gamma}(\alpha_0 = 0.001, \beta_0 = 0.001)
\end{align*}

\textbf{Posterior:} Normal-Inverse-Gamma (conjugate)
\begin{equation*}
p(\mu, \sigma^2 | \mathbf{r}) \propto p(\mathbf{r} | \mu, \sigma^2) \cdot p(\mu) \cdot p(\sigma^2)
\end{equation*}
\end{frame}

% Slide 3: MCMC - Rejection Sampling
\begin{frame}{MCMC Implementation: Rejection Sampling}
\textbf{Algorithm:}
\begin{enumerate}
    \item Define proposal distributions centered on data:
    \begin{align*}
    q(\mu) &= \mathcal{N}\left(\bar{r}, \sqrt{\sigma_0^2 + \frac{s^2}{n}}\right) \\
    q(\sigma^2) &= \text{Inv-Gamma}\left(\frac{n}{2}, \frac{1}{2}\sum_{i=1}^{n}(r_i - \bar{r})^2\right)
    \end{align*}
    
    \item Draw candidate $(\mu^*, \sigma^{2*})$ from proposal
    
    \item Accept with probability: $\alpha = \min\left(1, \frac{p^*(\mu^*, \sigma^{2*} | \mathbf{r})}{M \cdot q(\mu^*, \sigma^{2*})}\right)$
\end{enumerate}

\textbf{Implementation Details:}
\begin{itemize}
    \item Target samples: 2,000 | Max attempts: 1,000,000
    \item Envelope constant: $M = 10$
    \item Typical acceptance rate: 1-20\%
\end{itemize}
\end{frame}
